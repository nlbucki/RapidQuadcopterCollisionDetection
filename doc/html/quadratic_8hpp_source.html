74  inline std::pair<double, double>
75  quadraticEquation(const double a, const double b, const double c)
76  {
77  if (a == 0)
78  {
79  if (b == 0) throw NoQuadraticRoots();
80  double root = - c / b;
81  return std::make_pair(root, root);
82  }
83 
84  double discriminant = b * b - 4 * a * c;
85  if (discriminant < 0) throw NoQuadraticRoots();
86  double arg = std::sqrt(discriminant);
87  double q = -0.5 * ( b + ((b < 0) ? -arg : arg));
88 
89  return std::make_pair(q / a, c / q);
90  }

94  quadSolve(const double f0, const double f1, const double f2, double& root1, double& root2)
95  {
96  const double arg = f1 * f1 - 2 * f2 * f0;
97  if (arg < 0) return 0;
98  const double q = -(f1 + std::copysign(std::sqrt(arg), f1));
99  root1 = q / f2;
100  root2 = 2 * f0 / q;
101  return 2;
102  }
