203  void GetRotationMatrix(double R[9]) const {
204  const double r0 = _v[0] * _v[0];
205  const double r1 = _v[1] * _v[1];
206  const double r2 = _v[2] * _v[2];
207  const double r3 = _v[3] * _v[3];
208 
209  R[0] = r0 + r1 - r2 - r3;
210  R[1] = 2 * _v[1] * _v[2] - 2 * _v[0] * _v[3];
211  R[2] = 2 * _v[1] * _v[3] + 2 * _v[0] * _v[2];
212 
213  R[3] = 2 * _v[1] * _v[2] + 2 * _v[0] * _v[3];
214  R[4] = r0 - r1 + r2 - r3;
215  R[5] = 2 * _v[2] * _v[3] - 2 * _v[0] * _v[1];
216 
217  R[6] = 2 * _v[1] * _v[3] - 2 * _v[0] * _v[2];
218  R[7] = 2 * _v[2] * _v[3] + 2 * _v[0] * _v[1];
219  R[8] = r0 - r1 - r2 + r3;
