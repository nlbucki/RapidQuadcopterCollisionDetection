39  inline void
40  cubicNewtonRootPolish(const double& p, const double& q, const double& r, double& root, size_t iterations)
41  {
42  for (size_t it = 0; it < iterations; ++it)
43  {
44  double error = ((root + p) * root + q) * root + r;
45  double derivative = (3.0 * root + 2 * p) * root + q;
46 
47  if ((error == 0) || derivative == 0)
48  return;
49 
50  root -= error / derivative;
51  }
52  }
53  }

55  //Please read http://linus.it.uts.edu.au/~don/pubs/solving.html
56  //For solving cubics like x^3 + p * x^2 + q * x + r == 0
57  //This function always returns root1 >= root2 >= root3!
58  inline size_t
59  cubicSolve(const double& p, const double& q, const double& r, double& root1, double& root2, double& root3)
60  {
61  static const double maxSqrt = std::sqrt(std::numeric_limits<double>::max());
62 // static const double maxCubeRoot = std::pow(std::numeric_limits<double>::max(), 1.0 / 3.0); //Unused
63 
64  if (r == 0)
65  {
66  //no constant term, so divide by x and the result is a
67  //quadratic, but we must include the trivial x = 0 root
68  if (quadraticSolve(p, q, root1, root2))
69  {
70  root3 = 0;
71  if (root1 < root2)
72  std::swap(root1, root2);
73  if (root2 < 0)
74  {
75  std::swap(root2, root3);
76  if (root1 < 0)
77  std::swap(root1, root2);
78  }
79  return 3;
80  }
81  else
82  {
83  root1 = 0;
84  return 1;
85  }
86  }
87 
88  if ((p == 0) && (q == 0))
89  {
90  //Special case
91  //Equation is x^3 == -r
92  root1 = root2 = root3 = std::pow(-r, 1.0 / 3.0);
93  return 3;
94  }
95 
96  if ((p > maxSqrt) || (p < -maxSqrt))
97  {
98  //Equation limits to x^3 + p * x^2 == 0
99  root1 = -p;
100  return 1;
101  }
102 
103  if (q > maxSqrt)
104  {
105  //Special case, if q is large and the root is -r/q,
106  //The x^3 term is negligble, and all other terms cancel.
107  root1 = -r / q;
108  return 1;
109  }
110 
111  if (q < -maxSqrt)
112  {
113  //Special case, equation is x^3 + q x == 0
114  root1 = -std::sqrt(-q);
115  return 1;
116  }
117 
118  if ((r > maxSqrt) || (r < -maxSqrt))
119  {
120  //Another special case
121  //Equation is x^3 == -r
122  root1 = -std::pow(r, 1.0 / 3.0);
123  return 1;
124  }
125 
126  double v = r + (2.0 * p * p / 9.0 - q) * (p / 3.0);
127 
128  if ((v > maxSqrt) || (v < -maxSqrt))
129  {
130  root1 = -p;
131  return 1;
132  }
133 
134  double uo3 = q / 3.0 - p * p / 9.0;
135  double u2o3 = uo3 + uo3;
136 
137  if ((u2o3 > maxSqrt) || (u2o3 < -maxSqrt))
138  {
139  if (p == 0)
140  {
141  if (q > 0)
142  {
143  root1 = -r / q;
144  return 1;
145  }
146 
147  if (q < 0)
148  {
149  root1 = -std::sqrt(-q);
150  return 1;
151  }
152 
153  root1 = 0;
154  return 1;
155  }
156 
157  root1 = -q / p;
158  return 1;
159  }
160 
161  double uo3sq4 = u2o3 * u2o3;
162  if (uo3sq4 > maxSqrt)
163  {
164  if (p == 0)
165  {
166  if (q > 0)
167  {
168  root1 = -r / q;
169  return 1;
170  }
171 
172  if (q < 0)
173  {
174  root1 = -std::sqrt(-q);
175  return 1;
176  }
177 
178  root1 = 0;
179  return 1;
180  }
181 
182  root1 = -q / p;
183  return 1;
184  }
185 
186  double j = (uo3sq4 * uo3) + v * v;
187 
188  if (j > 0)
189  { //Only one root (but this test can be wrong due to a
190  //catastrophic cancellation in j
191  //(i.e. (uo3sq4 * uo3) == v * v)
192  double w = std::sqrt(j);
193  //double mcube;
194  if (v < 0)
195  root1 = std::pow(0.5 * (w - v), 1.0 / 3.0) - (uo3) * std::pow(2.0 / (w - v), 1.0 / 3.0) - p / 3.0;
196  else
197  root1 = uo3 * std::pow(2.0 / (w + v), 1.0 / 3.0) - std::pow(0.5 * (w + v), 1.0 / 3.0) - p / 3.0;
198 
199  //We now polish the root up before we use it in other calculations
200  detail::cubicNewtonRootPolish(p, q, r, root1, 15);
201 
202  //We double check that there are no more roots by using a
203  //quadratic formula on the factored problem, this helps when
204  //the j test is wrong due to numerical error.
205 
206  //We have a choice of either -r/root1, or q -
207  //(p+root1)*root1 for the constant term of the quadratic.
208  //
209  //The division one usually results in more accurate roots
210  //when it finds them but fails to detect real roots more
211  //often than the multiply.
212  if (quadraticSolve(p + root1, -r / root1, root2, root3))
213  return 3;
214  //
215  //However, the multiply detects roots where there are none,
216  //the division does not. So we must either accept missing
217  //roots or extra roots, here we choose missing roots
218  //
219  //if (quadSolve(q-(p+root1)*root1, p + root1, 1.0, root2, root3))
220  // return 3;
221 
222  return 1;
223  }
224 
225  if (uo3 >= 0)
226  { //Multiple root detected
227  root1 = root2 = root3 = std::pow(v, 1.0 / 3.0) - p / 3.0;
228  return 3;
229  }
230 
231  double muo3 = -uo3;
232  double s;
233  if (muo3 > 0)
234  {
235  s = std::sqrt(muo3);
236  if (p > 0)
237  s = -s;
238  }
239  else
240  s = 0;
241 
242  double scube = s * muo3;
243  if (scube == 0)
244  {
245  root1 = -p / 3.0;
246  return 1;
247  }
248 
249  double t = -v / (scube + scube);
250  double k = std::acos(t) / 3.0;
251  double cosk = std::cos(k);
252  root1 = (s + s) * cosk - p / 3.0;
253 
254  double sinsqk = 1.0 - cosk * cosk;
255  if (sinsqk < 0)
256  return 1;
257 
258  double rt3sink = std::sqrt(3.) * std::sqrt(sinsqk);
259  root2 = s * (-cosk + rt3sink) - p / 3.0;
260  root3 = s * (-cosk - rt3sink) - p / 3.0;
261 
262  detail::cubicNewtonRootPolish(p, q, r, root1, 15);
263  detail::cubicNewtonRootPolish(p, q, r, root2, 15);
264  detail::cubicNewtonRootPolish(p, q, r, root3, 15);
265 
266  return 3;
267  }
